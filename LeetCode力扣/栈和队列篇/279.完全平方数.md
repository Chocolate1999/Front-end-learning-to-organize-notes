给定正整数 n，找到若干个完全平方数（比如 1, 4, 9, 16, ...）使得它们的和等于 n。你需要让组成和的完全平方数的个数最少。

```
输入: n = 12
输出: 3 
解释: 12 = 4 + 4 + 4.
```

```
/**
 * @param {number} n
 * @return {number}
 */
var numSquares = function(n) {
    let queue = [],map = new Map()
    queue.push([n,0])
    map.set(n,false)
    while(queue.length){
        let [num,cnt] = queue.shift()
        for(let i=1;;i++){
            let cur = num-i*i
            if(cur<0) break
            else if(cur == 0) return cnt+1
            else{
                if(!map.get(cur)){
                    map.set(cur,true)
                    queue.push([cur,cnt+1])
                }
            }
        }
    }

};
```
